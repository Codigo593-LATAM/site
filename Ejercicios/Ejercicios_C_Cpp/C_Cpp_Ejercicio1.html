Aplicación de Factorial

Realizar un programa en C para calcular la siguiente serie:

S = 1/1! + 3/3! + 5/5! ... N/N! 
#include< stdio.h >

int main() {
    int N, i, j;
    float fac;
    float sum=0;
    printf("Ingrese valor de N:\n\t");
    scanf("%i", &N);
    for(i=1;i<=N;i=i+2){
        //calculo del factorial
        fac = 1;
        for(j=1;j<=i;j++){
            fac=fac*j;
        }
        sum = sum+i/fac;
    }
    printf("La respuesta es: %.4f", sum);
    return 0;
}
