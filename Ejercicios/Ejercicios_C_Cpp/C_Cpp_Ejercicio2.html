Aplicación en Trigonometría

Elaborar un programa al cual se le ingresa las coordenadas de dos puntos A y B (plano cartesiano) y calcula con dos decimales de aprocimación:
Las coordenadas del tercer únto C para formar un triángulo equilátero. Considerar que el tercer punto esta "arriba" de los dos puntos A y B.
El valor del lado del triángulo equilátero

Código

#include<math.h>
#include<stdio.h>
int main(){
    float Ax, Ay, Bx, By, Cx, Cy1, Cy2, Cy;
    float AB, AC, BC;
    float a,b,c;
    //Ax = 1.; Ay = 1.;
    //Bx = 8.; By = 2.;
    printf("Ingrese Punto A");
    printf("\nAx: ");
    scanf("%f", &Ax);
    printf("\nAy: " );
    scanf("%f", &Ay);

    printf("Ingrese Punto B");
    printf("\nBx");
    scanf("%f", &Bx);
    printf("\nBy" );
    scanf("%f", &By);
    //La distancias AB, AC, BC son las mismas
    AB = pow(pow(Bx - Ax,2)+pow(By-Ay,2), 0.5);
    AC = pow(pow(Cx - Ax,2)+pow(Cy-Ay,2), 0.5);
    
    a = pow((-2*By+2*Ay)/(-2*Bx+2*Ax),2)+1;
    b = (-2*(pow(Ay,2)-pow(By,2)+pow(Ax,2)-pow(Bx,2))*(-2*By+2*Ay))/(pow(-2*Bx+2*Ax,2))+2*Bx*(-2*By+2*Ay)/(-2*Bx+2*Ax)-2*By;
    c = pow((pow(Ay,2)-pow(By,2)+pow(Ax,2)-pow(Bx,2))/(-2*Bx+2*Ax),2)-2*Bx*(pow(Ay,2)-pow(By,2)+pow(Ax,2)-pow(Bx,2))/(-2*Bx+2*Ax)+pow(Bx,2)+pow(By,2)-pow(AB,2);
    Cy1 = (-b+pow(pow(b,2)-4*a*c,0.5))/(2*a);
    Cy2 = (-b-pow(pow(b,2)-4*a*c,0.5))/(2*a);
    if(Cy1>=Cy2){
        Cy = Cy1;
    }else{
        Cy = Cy2;
    }
    Cx = (pow(Ay,2)-pow(By,2)-Cy*(-2*By+2*Ay)+pow(Ax,2)-pow(Bx,2))/(-2*Bx+2*Ax);
    printf("\nC [%.2f , %.2f]", Cx, Cy);
    printf("\nlado %.2f", AB);
    return 0;
}
